```python
class EnsembleClassifier:
    """软投票集成：SVM + KNN + RF，按验证集 accuracy 加权投票"""
    def __init__(self, estimators: list, weights: list[float] | None = None):
        self.estimators = estimators  # [(name, clf), ...]
        self.weights = weights
        self.classes_ = None

    def fit(self, X, y):
        for _, clf in self.estimators:
            clf.fit(X, y)
        self.classes_ = self.estimators[0][1].classes_
        return self

    def predict(self, X):
        probas = self.predict_proba(X)
        return self.classes_[np.argmax(probas, axis=1)]

    def predict_proba(self, X):
        all_proba = []
        for _, clf in self.estimators:
            if hasattr(clf, "predict_proba"):
                all_proba.append(clf.predict_proba(X))
        avg = np.average(all_proba, axis=0, weights=self.weights)
        return avg
```

```python
    def __init__(self, bundle_path, use_cache=True, cache_maxsize=512):
        ...
        # LRU 缓存：key = img.tobytes() 的 hash → prediction result
        self._cache = {} if use_cache else None
        self._cache_maxsize = cache_maxsize

    def predict(self, img_bgr) -> dict:

        if self._cache is not None:
            if len(self._cache) >= self._cache_maxsize:
                # 简单 FIFO 淘汰：删除最早的一个
                self._cache.pop(next(iter(self._cache)))
            self._cache[cache_key] = result
        return result

    def predict_batch(self, imgs_bgr: list[np.ndarray]) -> list[dict]:
        """批量预测：一次性提取特征 → scaler → predict，比逐张快 3-5x"""
        # 用 builder.extract_batch 替代循环 extract_one
        # scaler.transform 一次处理整个 batch
        # classifier.predict 一次处理整个 batch
        X = self.builder.extract_batch(imgs_bgr).astype(np.float32)
        X_scaled = self.scaler.transform(X)
        y_preds = self.classifier.predict(X_scaled)
        results = []
        for i, y_pred in enumerate(y_preds):
            class_id = int(y_pred)
            confidence = None
            if hasattr(self.classifier, "predict_proba"):
                proba = self.classifier.predict_proba(X_scaled[i:i+1])
                confidence = float(proba[0, class_id])
            results.append({
                "class_id": class_id,
                "class_name": self.label_map.get(class_id, str(class_id)),
                "confidence": confidence,
            })
        return results

    @staticmethod
    def _img_hash(img) -> int:
        return hash(img.tobytes())
```

```python
class SignDetector:
    def __init__(self, predictor: Predictor,
                 scales=(0.5, 0.75, 1.0, 1.5, 2.0),
                 stride_ratio=0.1,
                 min_size=32, max_size_ratio=0.4,
                 iou_threshold=0.3,
                 confidence_threshold=0.4):
        """
        scales: 多尺度缩放因子
        stride_ratio: 滑动窗口步长占窗口尺寸比例
        min_size: 候选框最小边长
        max_size_ratio: 候选框最大边长占图像短边比例
        """
        ...

    def detect(self, img_bgr) -> list[dict]:
        """多尺度滑动窗口 + 颜色预筛选 + NMS"""
        all_candidates = []

        for scale in self.scales:
            # 1. 缩放图像
            scaled = cv2.resize(img_bgr, None, fx=scale, fy=scale)

            # 2. 颜色预筛选：HSV 掩膜生成候选区域
            mask = self._color_mask(scaled)
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

            # 3. 对每个轮廓扩展为候选框
            for cnt in contours:
                x, y, w, h = cv2.boundingRect(cnt)
                if w < self.min_size or h < self.min_size:
                    continue
                # 扩展框 20% 给上下文
                pad = int(max(w, h) * 0.2)
                x1 = max(0, x - pad)
                y1 = max(0, y - pad)
                x2 = min(scaled.shape[1], x + w + pad)
                y2 = min(scaled.shape[0], y + h + pad)
                crop = scaled[y1:y2, x1:x2]

                # 4. 预测
                result = self.predictor.predict(crop)
                if result["confidence"] and result["confidence"] >= self.confidence_threshold:
                    # 映射回原图坐标
                    orig_box = self._scale_box_back(x1, y1, x2-x1, y2-y1, scale)
                    all_candidates.append({
                        "bbox": orig_box,
                        "class_id": result["class_id"],
                        "class_name": result["class_name"],
                        "confidence": result["confidence"],
                    })

        # 5. NMS 去重
        return nms(all_candidates, self.iou_threshold)

    def _color_mask(self, img_bgr) -> np.ndarray:
        """HSV 颜色掩膜（红+蓝+黄），比原版增加黄色和更宽松的阈值"""
        hsv = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2HSV)
        # 红色两段
        mask_r1 = cv2.inRange(hsv, (0, 70, 50), (10, 255, 255))
        mask_r2 = cv2.inRange(hsv, (170, 70, 50), (180, 255, 255))
        # 蓝色
        mask_b = cv2.inRange(hsv, (90, 70, 50), (140, 255, 255))
        # 黄色（新增，覆盖警告标志）
        mask_y = cv2.inRange(hsv, (15, 70, 50), (35, 255, 255))
        mask = mask_r1 | mask_r2 | mask_b | mask_y
        # 形态学闭运算填充空洞
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel, iterations=2)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel, iterations=1)
        return mask
```

```python
        self.adaptive = adaptive

        if self.adaptive:
            img_bgr = self._adaptive_enhance(img_bgr)

    def _adaptive_enhance(self, img_bgr) -> np.ndarray:
        """根据图像质量自动调整增强参数"""
        gray = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY)
        brightness = gray.mean()
        contrast = gray.std()

        # 低光照：增强 CLAHE clip 限制
        if brightness < 80:
            self._clahe = cv2.createCLAHE(clipLimit=4.0, tileGridSize=(8,8))

        # 低对比度（雾天）：增加对比度

            l, a, b = cv2.split(lab)

            img_bgr = cv2.merge((l, a, b))

        # 运动模糊检测（Laplacian 方差 < 阈值时，尝试维纳去模糊）
```

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader

class TrafficSignDataset(Dataset):
    """从内存中的图像列表构建 PyTorch Dataset"""
    def __init__(self, images, labels, transform=None):
        self.images = images
        self.labels = labels
        self.transform = transform

    def __len__(self):
        return len(self.images)

    def __getitem__(self, idx):
        img = self.images[idx]
        # BGR → RGB → ToTensor
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        if self.transform:
            img = self.transform(img)
        else:
            img = torch.from_numpy(img).permute(2, 0, 1).float() / 255.0
        label = self.labels[idx]
        return img, label


class LeNet5(nn.Module):
    """经典 LeNet-5 适配 3 通道 64x64 输入"""
    def __init__(self, num_classes=43):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 6, 5), nn.ReLU(), nn.MaxPool2d(2),
            nn.Conv2d(6, 16, 5), nn.ReLU(), nn.MaxPool2d(2),
        )
        self.classifier = nn.Sequential(
            nn.Linear(16 * 13 * 13, 120), nn.ReLU(),
            nn.Linear(120, 84), nn.ReLU(),
            nn.Linear(84, num_classes),
        )

    def forward(self, x):
        x = self.features(x)
        x = x.view(x.size(0), -1)
        return self.classifier(x)


class MobileNetV3Small(nn.Module):
    """使用 torchvision 预训练 MobileNetV3-Small，替换分类头"""
    def __init__(self, num_classes=43):
        super().__init__()
        from torchvision import models
        self.backbone = models.mobilenet_v3_small(weights="DEFAULT")
        in_features = self.backbone.classifier[-1].in_features
        self.backbone.classifier[-1] = nn.Linear(in_features, num_classes)

    def forward(self, x):
        return self.backbone(x)


def train_cnn(model, train_loader, val_loader, epochs=20, lr=0.001, device="cuda"):
    """训练 CNN，返回最佳模型和训练历史"""
    model = model.to(device)
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    criterion = nn.CrossEntropyLoss()
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)

    best_val_acc = 0.0
    best_state = None
    history = {"train_loss": [], "val_acc": []}

    for epoch in range(epochs):
        model.train()
        total_loss = 0
        for images, labels in train_loader:
            images, labels = images.to(device), labels.to(device)
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()

        scheduler.step()

        # 验证
        val_acc = evaluate_cnn(model, val_loader, device)
        history["train_loss"].append(total_loss / len(train_loader))
        history["val_acc"].append(val_acc)

        if val_acc > best_val_acc:
            best_val_acc = val_acc
            best_state = {k: v.cpu().clone() for k, v in model.state_dict().items()}

    model.load_state_dict(best_state)
    return model, history, best_val_acc
```

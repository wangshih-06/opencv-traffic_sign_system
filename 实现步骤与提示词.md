```
**提示词**：

```
在 `traffic_sign_system/features/hog_extractor.py` 实现 HOG 提取器：

```python
class HOGExtractor:
    def __init__(self,
                 win_size=(64,64),
                 block_size=(16,16),
                 block_stride=(8,8),
                 cell_size=(8,8),
                 nbins=9):
        self.hog = cv2.HOGDescriptor(win_size, block_size, block_stride, cell_size, nbins)
        self.config = dict(win_size=win_size, block_size=block_size,
                           block_stride=block_stride, cell_size=cell_size, nbins=nbins)

    def extract(self, img_gray_u8: np.ndarray) -> np.ndarray:
        """返回一维 float32 特征向量"""
        feat = self.hog.compute(img_gray_u8)
        if feat is None:
            raise ValueError("HOG compute failed, image shape=%r" % (img_gray_u8.shape,))
        return feat.reshape(-1).astype(np.float32)

    def extract_batch(self, imgs_gray: list[np.ndarray]) -> np.ndarray:
        """返回 (N, D) 的 np.ndarray"""
        feats = [self.extract(im) for im in imgs_gray]
        return np.stack(feats, axis=0)

    def feature_dim(self) -> int:
        # 用一张 dummy 图跑一次得到 D
        dummy = np.zeros(self.config["win_size"][::-1], dtype=np.uint8)  # (h,w)
        return self.extract(dummy).shape[0]
```

要求：
- `extract` 失败必须抛清晰异常
- `feature_dim` 缓存到 `self._dim`，避免重复算

```
**目标**：计算 H/S 通道的二维直方图，固定维度（默认 8×8=64）。

**产出**：`traffic_sign_system/features/color_extractor.py`

**提示词**：

```
在 `traffic_sign_system/features/color_extractor.py` 实现 HSV 颜色直方图：

```python
class HSVColorHistogram:
    def __init__(self, h_bins: int = 8, s_bins: int = 8):
        self.h_bins = h_bins
        self.s_bins = s_bins
        self.config = dict(h_bins=h_bins, s_bins=s_bins)

    def extract(self, img_bgr: np.ndarray) -> np.ndarray:
        """返回 (h_bins*s_bins,) 的一维 L1 归一化直方图"""
        hsv = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2HSV)
        h, s, _ = cv2.split(hsv)
        # H 在 [0,180), S 在 [0,256)；按配置分箱
        hist = cv2.calcHist([hsv], channels=[0, 1], mask=None,
                            histSize=[self.h_bins, self.s_bins],
                            ranges=[0, 180, 0, 256])
        hist = hist.reshape(-1).astype(np.float32)
        hist /= (hist.sum() + 1e-6)
        return hist

    def extract_batch(self, imgs_bgr: list[np.ndarray]) -> np.ndarray:
        return np.stack([self.extract(im) for im in imgs_bgr], axis=0)
```

要求：
- L1 归一化（sum=1），便于和 HOG 拼接后做标准化
- 输入是 BGR（H,W,3），先转 HSV
- 给我一个验证：随机生成 10 张 (64,64,3) BGR 噪声图，extract_batch 后 shape=(10, 64)，每行 sum≈1
```

**自检**：每行 sum 接近 1.0（误差 < 1e-5）。
```

```python
class FeatureBuilder:
    def __init__(self, mode: str = "hog", img_size: int = 64,
                 h_bins: int = 8, s_bins: int = 8):
        """mode ∈ {"hog", "hsv", "hog+hsv"}"""
        assert mode in ("hog", "hsv", "hog+hsv")
        self.mode = mode
        self.hog = HOGExtractor(win_size=(img_size, img_size)) if mode != "hsv" else None
        self.hsv = HSVColorHistogram(h_bins, s_bins) if mode != "hog" else None
        self.config = dict(mode=mode, img_size=img_size, h_bins=h_bins, s_bins=s_bins)

    def extract_one(self, img_bgr: np.ndarray) -> np.ndarray:
        # 如果 mode 含 hog，先用 Preprocessor 走一遍得到灰度图
        if self.mode == "hog":
            gray = Preprocessor(to_gray=True, img_size=...)(img_bgr)
            return self.hog.extract(gray)
        if self.mode == "hsv":
            return self.hsv.extract(img_bgr)
        # hog+hsv：拼接
        gray = Preprocessor(...)(img_bgr)
        return np.concatenate([self.hog.extract(gray), self.hsv.extract(img_bgr)])

    def extract_batch(self, imgs_bgr):
        return np.stack([self.extract_one(im) for im in imgs_bgr], axis=0)
```

```python
from dataclasses import dataclass, field
import joblib, time, json
from pathlib import Path

@dataclass
class TrainSummary:
    model: str
    feature_mode: str
    n_train: int
    n_val: int
    n_test: int
    feature_dim: int
    train_seconds: float
    extras: dict = field(default_factory=dict)

REQUIRED_KEYS = ("classifier", "scaler", "label_map", "feature_config", "summary")

def save_bundle(out_path: Path, classifier, scaler, label_map, feature_config, summary: TrainSummary):
    out_path.parent.mkdir(parents=True, exist_ok=True)
    bundle = dict(classifier=classifier, scaler=scaler,
                  label_map=label_map, feature_config=feature_config,
                  summary=summary.__dict__)
    joblib.dump(bundle, out_path)

def load_bundle(path: Path) -> dict:
    bundle = joblib.load(path)
    missing = [k for k in REQUIRED_KEYS if k not in bundle]
    if missing:
        raise ValueError(f"模型包缺失字段: {missing}")
    return bundle
```

在 `traffic_sign_system/recognition/predictor.py` 实现统一预测器：

```python
class Predictor:
    def __init__(self, bundle_path: Path):
        bundle = load_bundle(bundle_path)
        self.classifier = bundle["classifier"]
        self.scaler = bundle["scaler"]
        self.label_map = bundle["label_map"]
        self.feature_config = bundle["feature_config"]
        self.builder = FeatureBuilder(**self.feature_config)  # 从 config 重建
        # 校验：scaler.mean_.shape[0] == builder.feature_dim
        ...

    def predict(self, img_bgr: np.ndarray) -> dict:
        X = self.builder.extract_one(img_bgr)[None, :]
        X = self.scaler.transform(X)
        y_pred = int(self.classifier.predict(X)[0])
        name = self.label_map[y_pred]
        prob = None
        if hasattr(self.classifier, "predict_proba"):
            p = float(self.classifier.predict_proba(X)[0, y_pred])
            prob = p
        return dict(class_id=y_pred, class_name=name, confidence=prob)
```
